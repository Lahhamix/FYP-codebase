// =========================
// Hardware pins
// =========================

// Interrupt pin from MAX30102 INT -> D2 on Uno
const byte oxiInt = 2;

// ADALOGGER pins (not used unless USE_ADALOGGER is defined)
#ifdef USE_ADALOGGER
  File dataFile;
  const byte chipSelect = 4;
  const byte cardDetect = 7;
  const byte batteryPin = 9;
  const byte ledPin = 13;        // Red LED on ADALOGGER
  const byte sdIndicatorPin = 8; // Green LED on ADALOGGER
  bool cardOK;
#endif

uint32_t elapsedTime, timeStart;

uint32_t aun_ir_buffer[BUFFER_SIZE];   // Infrared LED sensor data
uint32_t aun_red_buffer[BUFFER_SIZE];  // Red LED sensor data
float old_n_spo2;                      // Previous SpO2 value
uint8_t uch_dummy, k;

// Forward declarations
void millis_to_hours(uint32_t ms, char* hr_str);

#ifdef USE_ADALOGGER
void blinkLED(const byte led, bool isOK);
#endif

void setup() {

  // UNO CHANGE: INT is commonly active-low/open-drain -> use pullup for cleaner behavior
  pinMode(oxiInt, INPUT_PULLUP);

#ifdef USE_ADALOGGER
  pinMode(cardDetect, INPUT_PULLUP);
  pinMode(batteryPin, INPUT);
  pinMode(ledPin, OUTPUT);
  digitalWrite(ledPin, LOW);
  pinMode(sdIndicatorPin, OUTPUT);
  digitalWrite(sdIndicatorPin, LOW);
#endif

  // Serial is used in both DEBUG and non-ADALOGGER cases
  Serial.begin(115200);
  delay(200);

  Serial.println(F("Booting (Uno)..."));

  // Initialize MAX30102
  Serial.println(F("Reached before init"));
  maxim_max30102_init();
  Serial.println(F("Reached after init"));

  old_n_spo2 = 0.0;

#ifdef USE_ADALOGGER
  // ===== ADALOGGER SECTION (commented out by default for Uno) =====
  float measuredvbat = analogRead(batteryPin);
  measuredvbat *= 2;
  measuredvbat *= 3.3;
  measuredvbat /= 1024;

  char my_status[20];
  if (HIGH == digitalRead(cardDetect)) {
    if (!SD.begin(chipSelect)) {
      cardOK = false;
      strncpy(my_status, "CardInit!", 9);
    } else cardOK = true;
  } else {
    cardOK = false;
    strncpy(my_status, "NoSDCard!", 9);
  }

  if (cardOK) {
    long count = 0;
    char fname[20];
    do {
      sprintf(fname, "data_%ld.txt", ++count);
    } while (SD.exists(fname));
    dataFile = SD.open(fname, FILE_WRITE);
    strncpy(my_status, fname, 19);
  }

  blinkLED(ledPin, cardOK);

  k = 0;
  dataFile.println(F("Vbatt=\t"));
  dataFile.println(measuredvbat);
  dataFile.println(my_status);

#ifdef TEST_MAXIM_ALGORITHM
  dataFile.print(F("Time[s]\tSpO2\tHR\tSpO2_MX\tHR_MX\tClock\tRatio\tCorr\tTemp[C]"));
#else
  dataFile.print(F("Time[s]\tSpO2\tHR\tClock\tRatio\tCorr\tTemp[C]"));
#endif

#ifdef SAVE_RAW_DATA
  int32_t i;
  for (i = 0; i < BUFFER_SIZE; ++i) { dataFile.print("\t"); dataFile.print(i); }
  for (i = 0; i < BUFFER_SIZE; ++i) { dataFile.print("\t"); dataFile.print(i); }
#endif
  dataFile.println("");

#else
  // UNO CHANGE: Don’t block forever waiting for a key.
  // Optional: Give yourself 5 seconds to open Serial Monitor + send any char.
  Serial.println(F("Send any key within 5 seconds to start (or it will auto-start)..."));
  unsigned long t0 = millis();
  while (Serial.available() == 0 && (millis() - t0) < 5000UL) {
    delay(10);
  }
  if (Serial.available()) uch_dummy = Serial.read();

#ifdef TEST_MAXIM_ALGORITHM
  Serial.print(F("Time[s]\tSpO2\tHR\tSpO2_MX\tHR_MX\tClock\tRatio\tCorr\tTemp[C]"));
#else
  Serial.print(F("Time[s]\tSpO2\tHR\tClock\tRatio\tCorr\tTemp[C]"));
#endif

#ifdef SAVE_RAW_DATA
  int32_t i;
  for (i = 0; i < BUFFER_SIZE; ++i) { Serial.print("\t"); Serial.print(i); }
  for (i = 0; i < BUFFER_SIZE; ++i) { Serial.print("\t"); Serial.print(i); }
#endif
  Serial.println("");
#endif // USE_ADALOGGER

  timeStart = millis();
}

// Continuously taking samples from MAX30102.
// Heart rate and SpO2 are calculated every ST seconds.
void loop() {
  float n_spo2, ratio, correl;
  int8_t ch_spo2_valid;
  int32_t n_heart_rate;
  int8_t ch_hr_valid;
  int32_t i;
  char hr_str[10];

  // Read BUFFER_SIZE samples (ST seconds of samples running at FS sps)
  for (i = 0; i < BUFFER_SIZE; i++) {

    // Wait until interrupt asserts (goes LOW).
    // If you ever get "stuck" here, it means INT never fired -> wiring/config issue.
    while (digitalRead(oxiInt) == HIGH) {
      // Optional tiny delay to reduce CPU burn
      // delayMicroseconds(50);
    }

    maxim_max30102_read_fifo((aun_red_buffer + i), (aun_ir_buffer + i));

#ifdef DEBUG
    Serial.print(i);
    Serial.print(F("\t"));
    Serial.print(aun_red_buffer[i]);
    Serial.print(F("\t"));
    Serial.println(aun_ir_buffer[i]);
#endif
  }

  // Robert’s method
  rf_heart_rate_and_oxygen_saturation(
    aun_ir_buffer, BUFFER_SIZE,
    aun_red_buffer,
    &n_spo2, &ch_spo2_valid,
    &n_heart_rate, &ch_hr_valid,
    &ratio, &correl
  );

  elapsedTime = millis() - timeStart;
  millis_to_hours(elapsedTime, hr_str);
  elapsedTime /= 1000; // seconds

  // Read chip temperature
  int8_t integer_temperature;
  uint8_t fractional_temperature;
  maxim_max30102_read_temperature(&integer_temperature, &fractional_temperature);
  float temperature = integer_temperature + ((float)fractional_temperature) / 16.0;

#ifdef TEST_MAXIM_ALGORITHM
  float n_spo2_maxim;
  int8_t ch_spo2_valid_maxim;
  int32_t n_heart_rate_maxim;
  int8_t ch_hr_valid_maxim;

  maxim_heart_rate_and_oxygen_saturation(
    aun_ir_buffer, BUFFER_SIZE,
    aun_red_buffer,
    &n_spo2_maxim, &ch_spo2_valid_maxim,
    &n_heart_rate_maxim, &ch_hr_valid_maxim
  );
#endif

  // Print/save results when valid
#ifdef TEST_MAXIM_ALGORITHM
  if ((ch_hr_valid && ch_spo2_valid) || (ch_hr_valid_maxim && ch_spo2_valid_maxim)) {
#else
  if (ch_hr_valid && ch_spo2_valid) {
#endif

#ifdef USE_ADALOGGER
    ++k;
    dataFile.print(elapsedTime);   dataFile.print("\t");
    dataFile.print(n_spo2);        dataFile.print("\t");
    dataFile.print(n_heart_rate);  dataFile.print("\t");

#ifdef TEST_MAXIM_ALGORITHM
    dataFile.print(n_spo2_maxim);       dataFile.print("\t");
    dataFile.print(n_heart_rate_maxim); dataFile.print("\t");
#endif

    dataFile.print(hr_str);     dataFile.print("\t");
    dataFile.print(ratio);      dataFile.print("\t");
    dataFile.print(correl);     dataFile.print("\t");
    dataFile.print(temperature);

#ifdef SAVE_RAW_DATA
    for (i = 0; i < BUFFER_SIZE; ++i) { dataFile.print(F("\t")); dataFile.print(aun_red_buffer[i]); }
    for (i = 0; i < BUFFER_SIZE; ++i) { dataFile.print(F("\t")); dataFile.print(aun_ir_buffer[i]); }
#endif

    dataFile.println("");

    digitalWrite(sdIndicatorPin, HIGH);
    delay(10);
    digitalWrite(sdIndicatorPin, LOW);

    if (k >= 10) { dataFile.flush(); k = 0; }

#else
    Serial.print(elapsedTime);    Serial.print("\t");
    Serial.print(n_spo2);         Serial.print("\t");
    Serial.print(n_heart_rate);   Serial.print("\t");

#ifdef TEST_MAXIM_ALGORITHM
    Serial.print(n_spo2_maxim);       Serial.print("\t");
    Serial.print(n_heart_rate_maxim); Serial.print("\t");
#endif

    Serial.print(hr_str);        Serial.print("\t");
    Serial.print(ratio);         Serial.print("\t");
    Serial.print(correl);        Serial.print("\t");
    Serial.print(temperature);

#ifdef SAVE_RAW_DATA
    for (i = 0; i < BUFFER_SIZE; ++i) { Serial.print(F("\t")); Serial.print(aun_red_buffer[i]); }
    for (i = 0; i < BUFFER_SIZE; ++i) { Serial.print(F("\t")); Serial.print(aun_ir_buffer[i]); }
#endif

    Serial.println("");
#endif

    old_n_spo2 = n_spo2;
  }
}

void millis_to_hours(uint32_t ms, char* hr_str) {
  char istr[6];
  uint32_t secs, mins, hrs;
  secs = ms / 1000;
  mins = secs / 60;
  secs -= 60 * mins;
  hrs = mins / 60;
  mins -= 60 * hrs;

  itoa(hrs, hr_str, 10);
  strcat(hr_str, ":");
  itoa(mins, istr, 10);
  strcat(hr_str, istr);
  strcat(hr_str, ":");
  itoa(secs, istr, 10);
  strcat(hr_str, istr);
}

#ifdef USE_ADALOGGER
void blinkLED(const byte led, bool isOK) {
  byte i;
  if (isOK) {
    for (i = 0; i < 3; ++i) {
      digitalWrite(led, HIGH);
      delay(200);
      digitalWrite(led, LOW);
      delay(200);
    }
  } else {
    while (1) {
      for (i = 0; i < 2; ++i) {
        digitalWrite(led, HIGH);
        delay(50);
        digitalWrite(led, LOW);
        delay(50);
      }
      delay(500);
    }
  }
}
#endif
